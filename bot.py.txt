class Database:
    """SQLite Database Management Class"""
    
    def __init__(self, db_path: str):
        self.db_path = db_path
        self.init_db()
    
    def get_connection(self):
        """Get database connection"""
        conn = sqlite3.connect(self.db_path)
        conn.row_factory = sqlite3.Row
        return conn
    
    def init_db(self):
        """Initialize database tables"""
        conn = self.get_connection()
        cursor = conn.cursor()
        
        # Users table
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS users (
                user_id INTEGER PRIMARY KEY,
                username TEXT,
                first_name TEXT,
                join_date TEXT,
                invited_by INTEGER,
                referrals_count INTEGER DEFAULT 0,
                premium_status BOOLEAN DEFAULT 0,
                streak INTEGER DEFAULT 0,
                daily_bonus_date TEXT,
                generated_bios_count INTEGER DEFAULT 0,
                banned BOOLEAN DEFAULT 0,
                daily_bonus_used BOOLEAN DEFAULT 0,
                last_command_time REAL DEFAULT 0
            )
        ''')
        
        # Referrals table
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS referrals (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                referrer_id INTEGER,
                referred_id INTEGER,
                date TEXT,
                FOREIGN KEY(referrer_id) REFERENCES users(user_id),
                FOREIGN KEY(referred_id) REFERENCES users(user_id)
            )
        ''')
        
        # Generated content table
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS generated_content (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                user_id INTEGER,
                content_type TEXT,
                content TEXT,
                date TEXT,
                FOREIGN KEY(user_id) REFERENCES users(user_id)
            )
        ''')
        
        conn.commit()
        conn.close()
        logger.info("Database initialized successfully")
    
    def user_exists(self, user_id: int) -> bool:
        """Check if user exists"""
        conn = self.get_connection()
        cursor = conn.cursor()
        cursor.execute("SELECT user_id FROM users WHERE user_id = ?", (user_id,))
        result = cursor.fetchone()
        conn.close()
        return result is not None
    
    def add_user(self, user_id: int, username: str, first_name: str, invited_by: Optional[int] = None):
        """Add new user to database"""
        if self.user_exists(user_id):
            return
        
        conn = self.get_connection()
        cursor = conn.cursor()
        
        cursor.execute('''
            INSERT INTO users (user_id, username, first_name, join_date, invited_by)
            VALUES (?, ?, ?, ?, ?)
        ''', (user_id, username, first_name, datetime.now().isoformat(), invited_by))
        
        if invited_by:
            cursor.execute('''
                UPDATE users SET referrals_count = referrals_count + 1
                WHERE user_id = ?
            ''', (invited_by,))
            
            cursor.execute('''
                INSERT INTO referrals (referrer_id, referred_id, date)
                VALUES (?, ?, ?)
            ''', (invited_by, user_id, datetime.now().isoformat()))
        
        conn.commit()
        conn.close()
    
    def get_user(self, user_id: int) -> Optional[Dict]:
        """Get user data"""
        conn = self.get_connection()
        cursor = conn.cursor()
        cursor.execute("SELECT * FROM users WHERE user_id = ?", (user_id,))
        row = cursor.fetchone()
        conn.close()
        return dict(row) if row else None
    
    def get_referral_link(self, user_id: int) -> str:
        """Generate referral link for user"""
        return f"https://t.me/YOUR_BOT_USERNAME?start={user_id}"
    
    def get_referrals_count(self, user_id: int) -> int:
        """Get referral count for user"""
        user = self.get_user(user_id)
        return user['referrals_count'] if user else 0
    
    def get_top_inviters(self, limit: int = 10) -> List[Dict]:
        """Get top inviters list"""
        conn = self.get_connection()
        cursor = conn.cursor()
        cursor.execute('''
            SELECT user_id, username, first_name, referrals_count 
            FROM users 
            WHERE banned = 0
            ORDER BY referrals_count DESC 
            LIMIT ?
        ''', (limit,))
        results = cursor.fetchall()
        conn.close()
        return [dict(row) for row in results]
    
    def get_rank_tier(self, referrals: int) -> Tuple[str, str]:
        """Get rank tier based on referrals"""
        if referrals >= 50:
            return "🔱 LEGEND", "#FFD700"
        elif referrals >= 25:
            return "💎 DIAMOND", "#00FFFF"
        elif referrals >= 10:
            return "🏆 GOLD", "#FFD700"
        elif referrals >= 5:
            return "⭐ SILVER", "#C0C0C0"
        elif referrals >= 2:
            return "🥉 BRONZE", "#CD7F32"
        else:
            return "🌱 NOVICE", "#808080"
    
    def get_total_users(self) -> int:
        """Get total users count"""
        conn = self.get_connection()
        cursor = conn.cursor()
        cursor.execute("SELECT COUNT(*) FROM users")
        result = cursor.fetchone()[0]
        conn.close()
        return result
    
    def get_active_users(self, days: int = 7) -> int:
        """Get active users in last N days"""
        conn = self.get_connection()
        cursor = conn.cursor()
        since = datetime.now() - timedelta(days=days)
        cursor.execute('''
            SELECT COUNT(*) FROM users 
            WHERE join_date >= ?
        ''', (since.isoformat(),))
        result = cursor.fetchone()[0]
        conn.close()
        return result
    
    def give_premium(self, user_id: int, days: int = 30):
        """Give premium status to user"""
        conn = self.get_connection()
        cursor = conn.cursor()
        cursor.execute('''
            UPDATE users 
            SET premium_status = 1
            WHERE user_id = ?
        ''', (user_id,))
        conn.commit()
        conn.close()
    
    def remove_premium(self, user_id: int):
        """Remove premium status from user"""
        conn = self.get_connection()
        cursor = conn.cursor()
        cursor.execute('''
            UPDATE users 
            SET premium_status = 0
            WHERE user_id = ?
        ''', (user_id,))
        conn.commit()
        conn.close()
    
    def ban_user(self, user_id: int):
        """Ban user"""
        conn = self.get_connection()
        cursor = conn.cursor()
        cursor.execute('''
            UPDATE users 
            SET banned = 1
            WHERE user_id = ?
        ''', (user_id,))
        conn.commit()
        conn.close()
    
    def unban_user(self, user_id: int):
        """Unban user"""
        conn = self.get_connection()
        cursor = conn.cursor()
        cursor.execute('''
            UPDATE users 
            SET banned = 0
            WHERE user_id = ?
        ''', (user_id,))
        conn.commit()
        conn.close()
    
    def is_banned(self, user_id: int) -> bool:
        """Check if user is banned"""
        user = self.get_user(user_id)
        return user['banned'] if user else False
    
    def save_generated_content(self, user_id: int, content_type: str, content: str):
        """Save generated content"""
        conn = self.get_connection()
        cursor = conn.cursor()
        cursor.execute('''
            INSERT INTO generated_content (user_id, content_type, content, date)
            VALUES (?, ?, ?, ?)
        ''', (user_id, content_type, content, datetime.now().isoformat()))
        
        cursor.execute('''
            UPDATE users 
            SET generated_bios_count = generated_bios_count + 1
            WHERE user_id = ?
        ''', (user_id,))
        
        conn.commit()
        conn.close()
    
    def check_daily_bonus_available(self, user_id: int) -> bool:
        """Check if daily bonus is available"""
        user = self.get_user(user_id)
        if not user:
            return False
        
        if user['daily_bonus_date'] is None:
            return True
        
        last_bonus = datetime.fromisoformat(user['daily_bonus_date'])
        return (datetime.now() - last_bonus).days >= 1
    
    def claim_daily_bonus(self, user_id: int):
        """Claim daily bonus"""
        conn = self.get_connection()
        cursor = conn.cursor()
        cursor.execute('''
            UPDATE users 
            SET daily_bonus_date = ?, streak = streak + 1
            WHERE user_id = ?
        ''', (datetime.now().isoformat(), user_id))
        conn.commit()
        conn.close()
    
    def reset_daily_bonus(self, user_id: int):
        """Reset daily bonus streak if missed a day"""
        user = self.get_user(user_id)
        if user and user['daily_bonus_date']:
            last_bonus = datetime.fromisoformat(user['daily_bonus_date'])
            if (datetime.now() - last_bonus).days > 1:
                conn = self.get_connection()
                cursor = conn.cursor()
                cursor.execute('''
                    UPDATE users 
                    SET streak = 0
                    WHERE user_id = ?
                ''', (user_id,))
                conn.commit()
                conn.close()
    
    def get_bot_stats(self) -> Dict:
        """Get bot statistics"""
        conn = self.get_connection()
        cursor = conn.cursor()
        
        cursor.execute("SELECT COUNT(*) FROM users WHERE banned = 0")
        total_users = cursor.fetchone()[0]
        
        cursor.execute("SELECT COUNT(*) FROM users WHERE premium_status = 1")
        premium_users = cursor.fetchone()[0]
        
        cursor.execute("SELECT COUNT(*) FROM generated_content")
        total_generated = cursor.fetchone()[0]
        
        cursor.execute("SELECT SUM(referrals_count) FROM users")
        total_referrals = cursor.fetchone()[0] or 0
        
        cursor.execute('''
            SELECT COUNT(*) FROM users 
            WHERE join_date >= datetime('now', '-7 days')
        ''')
        new_users_week = cursor.fetchone()[0]
        
        conn.close()
        
        return {
            'total_users': total_users,
            'premium_users': premium_users,
            'total_generated': total_generated,
            'total_referrals': total_referrals,
            'new_users_week': new_users_week
        }

class AntiSpamMiddleware:
    """Anti-spam middleware to prevent command spamming"""
    
    def __init__(self, time_limit: float = 0.5):
        self.time_limit = time_limit
        self.last_commands = {}
    
    async def __call__(self, handler, event, data):
        from time import time
        
        if not isinstance(event, Message):
            return await handler(event, data)
        
        user_id = event.from_user.id
        current_time = time()
        
        if user_id in self.last_commands:
            if current_time - self.last_commands[user_id] < self.time_limit:
                return
        
        self.last_commands[user_id] = current_time
        return await handler(event, data)

    username = message.from_user.username or "Unknown"
    first_name = message.from_user.first_name or "User"
    
    # Check if user is banned
    if db.is_banned(user_id):
        await message.reply("⛔ You have been banned from using this bot.")

    invited_by = None
    args = message.text.split()
    
    if len(args) > 1 and args[1].isdigit():
        invited_by = int(args[1])
        if not db.user_exists(invited_by):
            invited_by = None
    
    # Add user to database
    if not db.user_exists(user_id):
        db.add_user(user_id, username, first_name, invited_by)
        logger.info(f"New user added: {user_id} ({first_name})")
    
    # Send welcome message
    welcome_text = f"""

✅ Telegram

👥 Invite friends → Unlock elite styles
💎 More invites → Higher rank
👑 Become LEGEND!

{get_referral_progress_text(db.get_referrals_count(user_id))}

💪 Ready? Let's go!

async def handle_admin_action(message: Message, state: FSMContext):
    """Handle admin action (give/remove premium, ban/unban)"""
    user_id = message.from_user.id
    
    if user_id not in ADMIN_IDS:
        return
    
    try:
        target_id = int(message.text)
        current_state = await state.get_state()
        
        if "give" in current_state.lower() or "premium" in current_state.lower():
            db.give_premium(target_id)
            await message.reply(f"✅ Premium given to user {target_id}")
        elif "ban" in current_state.lower():
            db.ban_user(target_id)
            await message.reply(f"⛔ User {target_id} has been banned")
        elif "remove" in current_state.lower():
            db.remove_premium(target_id)
            await message.reply(f"🎁 Premium removed from user {target_id}")
        elif "unban" in current_state.lower():
            db.unban_user(target_id)
            await message.reply(f"✅ User {target_id} has been unbanned")
        
        await state.clear()
    
    except ValueError:
        await message.reply("❌ Invalid user ID. Please send a valid number.")

async def handle_broadcast_text(message: Message, state: FSMContext):
    """Handle broadcast message"""
    user_id = message.from_user.id
    
    if user_id not in ADMIN_IDS:
        return
    
    # Get all users
    conn = db.get_connection()
    cursor = conn.cursor()
    cursor.execute("SELECT user_id FROM users WHERE banned = 0")
    users = cursor.fetchall()
    conn.close()
    
    sent = 0
    failed = 0
    
    for user_row in users:
        try:
            await message.bot.send_message(
                user_row[0],
                f"📢 {bold('ANNOUNCEMENT')} 📢\n\n" + message.text,
                parse_mode=ParseMode.HTML
            )
            sent += 1
        except:
            failed += 1
    
    await message.reply(f"✅ Broadcast complete!\n📤 Sent: {sent}\n❌ Failed: {failed}")
    await state.clear()

async def main():
    """Main bot function"""
    global db
    
    # Initialize database
    db = Database(DATABASE_PATH)
    
    # Initialize bot and dispatcher
    bot = Bot(token=BOT_TOKEN, parse_mode=ParseMode.HTML)
    dp = Dispatcher()
    
    # Setup default commands
    await setup_default_commands(bot)
    
    # Register routers
    router = Router()
    
    # Commands
    router.message.register(cmd_start, Command("start"))
    router.message.register(cmd_help, Command("help"))
    router.message.register(cmd_profile, Command("profile"))
    router.message.register(cmd_referral, Command("referral"))
    router.message.register(cmd_leaderboard, Command("leaderboard"))
    router.message.register(cmd_admin, Command("admin"))
    
    # Main menu
    router.message.register(handle_main_menu)
    
    # Admin panel
    router.message.register(
        handle_admin_text,
        StateFilter(AdminStates.admin_managing)
    )
    router.message.register(
        handle_admin_action,
        StateFilter(AdminStates.waiting_message)
    )
    router.message.register(
        handle_broadcast_text,
        StateFilter(AdminStates.waiting_broadcast_type)
    )
    
    # Callbacks
    router.callback_query.register(handle_callback)
    
    dp.include_router(router)
    
    # Start polling
    logger.info("Bot started successfully!")
    try:
        await dp.start_polling(bot)
    except KeyboardInterrupt:
        logger.info("Bot stopped")
    finally:
        await bot.session.close()

if __name__ == "__main__":
    asyncio.run(main())
